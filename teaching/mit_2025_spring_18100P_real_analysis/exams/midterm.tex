    \textbf{Foundations / Logic}
    \begin{itemize}
        \item (6 Pts)
            Write down the truth table for:
            \begin{enumerate}
                \item
                    $P\Rightarrow{Q}$
                \item
                    $P\Rightarrow\neg{Q}$
                \item
                    $(P\Rightarrow\neg{Q})\Rightarrow\neg{P}$
                \item
                    $(P\Rightarrow{Q})\Rightarrow\big((P\Rightarrow\neg{Q})\Rightarrow\neg{P}\big)$
            \end{enumerate}
        \item (4 Pts)
            Is it consistent to accept this final expression
            as an axiom of logic? Explain using your truth table.
    \end{itemize}
    \newpage
    \textbf{Sequences and Subsequences}
    \begin{itemize}
        \item (2 Pts)
            Provide the precise definition of a convergent sequence of real
            numbers.
        \item (4 Pts)
            Prove that if $a:\mathbb{N}\rightarrow\mathbb{R}$ is a
            convergent sequence, and if $k:\mathbb{N}\rightarrow\mathbb{N}$
            is strictly increasing, then the subsequence $a_{k}$ also
            converges.
        \item (4 Pts)
            Prove or refute the claim that
            $a_{n}=\sin(n)$ has a convergent subsequence
            (you may use any of the theorems from class without proving them).
    \end{itemize}
    \newpage
    \textbf{Cauchy Sequences}
    \begin{itemize}
        \item (2 Pts)
            Provide the precise definition of a Cauchy sequence of real numbers.
        \item
            (4 Pts)
            Prove or provide a counterexample to the following:
            If $a:\mathbb{N}\rightarrow\mathbb{R}$ is a Cauchy sequence, then
            $a_{n+1}-a_{n}\rightarrow{0}$.
        \item
            (4 Pts)
            Prove or provide a counterexample to the following:
            If $a:\mathbb{N}\rightarrow\mathbb{R}$ is a sequence and
            $a_{n+1}-a_{n}\rightarrow{0}$, then $a$ is a Cauchy sequence.
    \end{itemize}
    \newpage
    \textbf{Compactness, Open and Closed Sets}
    \begin{itemize}
        \item (6 Pts)
            Prove that if
            $\mathcal{C}:\mathbb{N}\rightarrow\mathcal{P}(\mathbb{R})$ is a
            sequence of closed sets (that is, $\mathcal{C}_{n}$ is a closed
            subset of $\mathbb{R}$ for each $n\in\mathbb{N}$), then
            $\bigcap_{n=0}^{\infty}\mathcal{C}_{n}$ is closed
            (this result holds for arbitrary families of closed sets, not
            just sequences of them, but you only need to prove the claim
            for a sequence of closed sets).
        \item
            (2 Pts)
            Recall that if $A$ and $B$ are subsets of $\mathbb{R}$, then
            the set difference $B\setminus{A}$ is given by:
            \begin{equation}
                B\setminus{A}
                =
                B\cap{A}^{C}
                =
                B\cap\left(\mathbb{R}\setminus{A}\right)
            \end{equation}
            Here $A^{C}$ means the \textit{complement} of $A$ in $\mathbb{R}$.

            Prove that if $\mathcal{C}\subseteq\mathbb{R}$ is closed and
            $\mathcal{U}\subseteq\mathbb{R}$ is open, then
            $\mathcal{C}\setminus\mathcal{U}$ is closed
            (Hint: Don't overthink it, use what you've already proven).
        \item
            (2 Pts)
            Let $\mathcal{C}_{0}=[0,\,1]$, the closed unit interval.
            Let $\mathcal{C}_{1}=[0,\,1/3]\cup[2/3,\,1]$, the closed unit
            interval with the \textit{open middle thirds} removed.
            If $\mathcal{C}_{n}$ is defined, define $\mathcal{C}_{n+1}$ from
            $\mathcal{C}_{n}$ by removing all of the
            \textit{open middle thirds} from each of the closed
            subintervals. Prove that $\bigcap_{n=0}^{\infty}\mathcal{C}_{n}$
            is compact
            (Hint: You may use any of the major results from class freely).
    \end{itemize}
    \newpage
    \textbf{Continuity}
    \begin{itemize}
        \item (10 Pts)
            Prove using an $\varepsilon-\delta$ argument that
            $f:\mathbb{R}\rightarrow\mathbb{R}$ given by:
            \begin{equation}
                f(x)
                =
                \begin{cases}
                    x,&x\in\mathbb{Q}\\
                    -x,&x\not\in\mathbb{Q}
                \end{cases}
            \end{equation}
            is continuous at $x=0$ and discontinuous everywhere else.
    \end{itemize}
