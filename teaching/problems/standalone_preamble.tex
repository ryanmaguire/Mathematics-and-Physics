    colorlinks=true,
    linkcolor=blue
}

\newif\ifShowSolution

% Theorem style for default spacing and normal font. Used for problems.
\newtheoremstyle{normal}
    {\topsep}   % Amount of space above the theorem.
    {\topsep}   % Amount of space below the theorem.
    {}          % Font used for body of theorem.
    {}          % Measure of space to indent.
    {\bfseries} % Font of the header of the theorem.
    {}          % Punctuation between head and body.
    {.5em}      % Space after theorem head.
    {}

% Theorem style for default spacing and italic font. Used for solutions.
\newtheoremstyle{italic}
    {\topsep}   % Amount of space above the theorem.
    {\topsep}   % Amount of space below the theorem.
    {}          % Font used for body of theorem.
    {}          % Measure of space to indent.
    {\itshape}  % Font of the header of the theorem.
    {}          % Punctuation between head and body.
    {.5em}      % Space after theorem head.
    {Solution.} % Text that appears at the start of the environment.

% Define default environments.
\theoremstyle{normal}
\newtheorem{problem}{Problem}
\theoremstyle{italic}
\newtheorem{solution}{Solution}
