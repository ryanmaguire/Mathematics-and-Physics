%   Unary minus sign                                                           %

%-------------------------------Main Document----------------------------------%
\begin{tikzpicture}[%
    >=Latex,
    line width=0.2mm,
    line cap=round
]
    % Coordinates for various points.
    \coordinate (P1)    at (-1.0, -2.0);
    \coordinate (P2)    at ( 2.0,  3.0);
    \coordinate (START) at (-2.2, -4.0);
    \coordinate (END)   at ( 2.6,  4.0);

    % Draw a grid.
    \draw[style=help lines] (-3.9, -3.9) grid (3.9, 3.9);

    % Axes:
    \begin{scope}[very thick]
        \draw[<->] (-4.1,  0.0) to (4.1, 0.0) node [above] {$x$};
        \draw[<->] ( 0.0, -4.1) to (0.0, 4.1) node [right] {$y$};
    \end{scope}

    % Label the axes.
    \foreach\n in {1, 2, 3}{%
        \draw (\n,  3pt) to (\n,  -3pt)  node [below] {$\n$};
        \draw (3pt,  \n) to (-3pt,  \n)  node [left]  {$\n$};
        \draw (-\n, 3pt) to (-\n,  -3pt) node [below] {$\minus\n$};
        \draw (3pt, -\n) to (-3pt, -\n)  node [left]  {$\minus\n$};
    }

    % Draw the line through the two points.
    \draw (START) to (END);

    % Label the two points.
    \draw[fill=black] (P1) circle (0.7mm) node [above left]  {$(\minus{1},\,\minus{2})$};
    \draw[fill=black] (P2) circle (0.7mm) node [below right] {$(2,\,3)$};
\end{tikzpicture}
