%   Use the standalone class for displaying the tikz image on a small PDF.
\documentclass[crop, tikz]{standalone}

%   Import the tikz package to use for the drawing.
\usepackage{tikz}

%   Begin the document.
\begin{document}

    %   Draw the figure.
    \begin{tikzpicture}[%
        line width=0.6pt,
        line cap=round
    ]
        %   Use this scope to mark circles for the start and end of the circuit.
        \begin{scope}[
            every node/.style={
                circle,
                fill=white,
                draw=black,
                inner sep=0pt,
                minimum size=3pt
            }
        ]
            \node (a) at (0.0, 0.0) {};
            \node (b) at (4.5, 0.0) {};
        \end{scope}

        %   Draw all parts of the circuit.
        \draw (a) to (1.0, 0.0);
        \draw (1.0, -1.0) to (1.0, 1.0);
        \draw (1.0, 1.0) to (2.0, 1.0);
        \draw (1.0, -1.0) to (2.0, -1.0);
        \draw (2.0, 0.5) to (2.0, 1.5);
        \draw (2.0, -0.5) to (2.0, -1.5);
        \draw (2.5, 0.5) to (2.5, 1.5);
        \draw (2.5, -0.5) to (2.5, -1.5);
        \draw (2.5, 1.0) to (3.5, 1.0);
        \draw (2.5, -1.0) to (3.5, -1.0);
        \draw (3.5, -1.0) to (3.5, 1.0);
        \draw (3.5, 0.0) to (b);

        %   Label the plates.
        \node at (2.25, 1.5) {\scriptsize{$C_{1}$}};
        \node at (2.25, -1.5) {\scriptsize{$C_{2}$}};
    \end{tikzpicture}
\end{document}
