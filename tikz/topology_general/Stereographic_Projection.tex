\begin{tikzpicture}[>=Latex]

    % Coordinates for all of the points.
    \coordinate (O)   at ( 0.00,  0.00);
    \coordinate (y)   at ( 2.00,  0.00);
    \coordinate (z)   at ( 0.00,  3.00);
    \coordinate (x)   at (-1.00, -1.00);
    \coordinate (xy)  at ( 1.00, -1.00);
    \coordinate (xyz) at ( 1.00,  0.25);
    \coordinate (Y)   at ( 3.42,  0.00);
    \coordinate (X)   at (-1.71, -1.71);
    \coordinate (XY)  at ( 1.71, -1.71);

    % Draw the main coordinate system axes
    \begin{scope}[thick, ->]
        \draw (O) to ( 4.0,  0.0) node[above] {$y$};
        \draw (O) to (-2.0, -2.0) node[left]  {$x$};
        \draw (O) to ( 0.0,  4.0) node[right] {$z$};
    \end{scope}
    
    % Draw blue line "inside" of sphere.
    \draw[draw=blue, semithick] (z) to (xyz);

    % Draw sphere.
    \draw[fill=gray, opacity=0.5] (0:3) arc(0:-180:3 and 1.3) arc(180:0:3);
    \draw (0:3) arc(0:-180:3 and 1.3) arc(180:0:3);
    \draw (90:3) arc(120:180:4.46);

    % Draw blue line "outside" of sphere.
    \draw[draw=blue, semithick, ->, shorten >=0.13cm] (xyz) to (XY);

    % Draw projection lines.
    \begin{scope}[dashed]
        \draw (O)  to (XY);
        \draw (X)  to (XY);
        \draw (Y)  to (XY);
        \draw (x)  to (xy);
        \draw (y)  to (xy);
        \draw (xy) to (xyz);
    \end{scope}

    % Draw the position vector.
    \draw[dashed,semithick] (O) to (xyz);
    \draw[fill=black] (xyz) circle (0.5mm);
    \draw[fill=black] (XY)  circle (0.5mm);
    \node at (xyz) [above right] {$(x,y,z)$};
    \node at (XY)  [below right]  {$(X,Y)$};
\end{tikzpicture}
