            in {25,24,...,2}{%
                \foreach\Y in {1,...,\Ymax}{%
                    \ifnum\X<5

                        (\Y/\X,0.02) -- (\Y/\X,-0.02)

                    \else
                        \draw[ultra thin]
                        (\Y/\X,0.01) to (\Y/\X,-0.01);
                    \fi
                    \pgfmathtruncatemacro{\TST}
                        {gcd(\X,\Y)}
                    \ifnum\TST=1
                        \fill ({\Y/\X},1/\X) 
                            circle (0.2pt); 
                    \fi
                }
        }
