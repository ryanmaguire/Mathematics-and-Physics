\begin{tikzpicture}[%
    ->-/.style={%
        decoration={%
            markings,
            mark=at position .55 with \arrow{Stealth}
        },
        postaction={decorate}
    }
]
    \coordinate (e)  at (0,0);
    \coordinate (r)  at (6,0);
    \coordinate (r2) at (6,6);
    \coordinate (r3) at (0,6);

    \coordinate (f)   at (2,2);
    \coordinate (fr)  at (2,4);
    \coordinate (fr2) at (4,4);
    \coordinate (fr3) at (4,2);

    % Dots for the coordinates.
    \draw[fill=black] (e)   circle (0.05);
    \draw[fill=black] (r)   circle (0.05);
    \draw[fill=black] (r2)  circle (0.05);
    \draw[fill=black] (r3)  circle (0.05);
    \draw[fill=black] (f)   circle (0.05);
    \draw[fill=black] (fr)  circle (0.05);
    \draw[fill=black] (fr2) circle (0.05);
    \draw[fill=black] (fr3) circle (0.05);

    % Solid arrows for rotations.
    \draw[->-] (e)   to (r);
    \draw[->-] (r)   to (r2);
    \draw[->-] (r2)  to (r3);
    \draw[->-] (r3)  to (e);
    \draw[->-] (f)   to (fr);
    \draw[->-] (fr)  to (fr2);
    \draw[->-] (fr2) to (fr3);
    \draw[->-] (fr3) to (f);

    % Dashed arrows for reflections.
    \draw[densely dashed] (e) to (f);
    \draw[densely dashed] (r) to (fr3);
    \draw[densely dashed] (r2) to (fr2);
    \draw[densely dashed] (r3) to (fr);

    \node at (e)   [below left]  {$e$};
    \node at (r)   [below right] {$r$};
    \node at (r2)  [above right] {$r^{2}$};
    \node at (r3)  [above left]  {$r^{3}$};
    \node at (f)   [above right] {$f$};
    \node at (fr3) [above left]  {$fr^{3}$};
    \node at (fr2) [below left]  {$fr^{2}$};
    \node at (fr)  [below right] {$fr$};
\end{tikzpicture}
