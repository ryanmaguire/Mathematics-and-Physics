\begin{tikzpicture}[%
    >=Latex,
    line width=0.2mm,
    line cap=round,
    font=\large
]

    % Coordinates for various points.
    \coordinate (O)   at (0.0, 0.0);
    \coordinate (z)   at (30:2in);
    \coordinate (x)   at (1.5in, 0.0in);
    \coordinate (C)   at (1.9in, 1.9in);

    % Axes:
    \begin{scope}[thick]
        \draw[->] (-0.2in,  0.0in) to (2.5in, 0.0in) node [above] {$\Re\{z\}$};
        \draw[->] ( 0.0in, -0.2in) to (0.0in, 2.5in) node [right] {$\Im\{z\}$};
    \end{scope}

    % Axes labels:
    \foreach\n in {1, 2}{%
        \draw (\n{in}, 3pt) to (\n{in}, -3pt) node [below] {$\n$};
        \draw (3pt, \n{in}) to (-3pt, \n{in}) node [left] {$\n{i}$};
    }

    % Draw a line from the origin to the point z.
    \draw (O) to node [above] {$r$} (z);

    % Draw a dot marking the point z.
    \draw[fill=black] (z) circle (0.4mm);

    % Nodes to label various things.
    \node at (C) {\Large{$\mathbb{C}$}};
    \node at (z) [above] {$z=(r,\theta)$};
    \node at (x) [above right=0.3in and 0.1cm] {$\theta$};
    \draw (x) arc (0:30:1.5in);
\end{tikzpicture}
